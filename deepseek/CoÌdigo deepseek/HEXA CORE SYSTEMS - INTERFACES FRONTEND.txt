import React, { useState, useEffect, useCallback } from 'react';
import axios from '../utils/axios-instance';
import { useSafeData } from '../hooks/useSafeData';
import { ErrorMessage } from '../components/ui/ErrorMessage';
import { LoadingSpinner } from '../components/ui/LoadingSpinner';

interface ModuleLicense {
  id: string;
  erpActive: boolean;
  posActive: boolean;
  hrActive: boolean;
  billingActive: boolean;
  logisticsActive: boolean;
  manufacturingActive: boolean;
  treasuryActive: boolean;
  reportsActive: boolean;
}

interface ModuleToggle {
  key: keyof Omit<ModuleLicense, 'id' | 'updatedAt'>;
  label: string;
  description: string;
  icon: string;
}

const MODULES: ModuleToggle[] = [
  {
    key: 'erpActive',
    label: 'ERP Core',
    description: 'Sistema central de gestión empresarial',
    icon: '🏭',
  },
  {
    key: 'posActive',
    label: 'Punto de Venta',
    description: 'Módulo de ventas y facturación en mostrador',
    icon: '🛒',
  },
  {
    key: 'hrActive',
    label: 'Recursos Humanos',
    description: 'Gestión de empleados, nómina y comisiones',
    icon: '👥',
  },
  {
    key: 'billingActive',
    label: 'Facturación Electrónica',
    description: 'CFDI, timbrado y complementos de pago',
    icon: '🧾',
  },
  {
    key: 'logisticsActive',
    label: 'Logística y Almacenes',
    description: 'Traspasos, inventarios multi-almacén',
    icon: '🚚',
  },
  {
    key: 'manufacturingActive',
    label: 'Manufactura',
    description: 'Recetas, BOM y órdenes de producción',
    icon: '⚙️',
  },
  {
    key: 'treasuryActive',
    label: 'Tesorería',
    description: 'Cuentas bancarias, movimientos y conciliación',
    icon: '🏦',
  },
  {
    key: 'reportsActive',
    label: 'Reportes Avanzados',
    description: 'Dashboard ejecutivo y reportes personalizados',
    icon: '📊',
  },
];

const SuperAdminPanel: React.FC = () => {
  const [license, setLicense] = useState<ModuleLicense | null>(null);
  const [loading, setLoading] = useState<boolean>(true);
  const [saving, setSaving] = useState<string | null>(null);
  const [error, setError] = useState<string | null>(null);
  const [successMessage, setSuccessMessage] = useState<string | null>(null);

  const fetchLicense = useCallback(async () => {
    try {
      setLoading(true);
      setError(null);
      const response = await axios.get('/api/superadmin/license');
      setLicense(response.data);
    } catch (err: any) {
      const message = err.response?.data?.error || 'Error al cargar la licencia';
      setError(message);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    fetchLicense();
  }, [fetchLicense]);

  const handleToggle = async (moduleKey: keyof Omit<ModuleLicense, 'id' | 'updatedAt'>) => {
    if (!license) return;

    const currentValue = license[moduleKey];
    const newValue = !currentValue;

    // Optimistic update
    setLicense((prev) => (prev ? { ...prev, [moduleKey]: newValue } : prev));
    setSaving(moduleKey);
    setError(null);
    setSuccessMessage(null);

    try {
      const payload = { [moduleKey]: newValue };
      const response = await axios.put('/api/superadmin/license', payload);
      setLicense(response.data);
      setSuccessMessage(`Módulo ${MODULES.find((m) => m.key === moduleKey)?.label} ${newValue ? 'activado' : 'desactivado'} correctamente`);
      setTimeout(() => setSuccessMessage(null), 3000);
    } catch (err: any) {
      // Revertir cambio optimista
      setLicense((prev) => (prev ? { ...prev, [moduleKey]: currentValue } : prev));
      const message = err.response?.data?.error || 'Error al actualizar licencia';
      setError(message);
    } finally {
      setSaving(null);
    }
  };

  const activeCount = license
    ? MODULES.filter((m) => license[m.key]).length
    : 0;

  if (loading) {
    return (
      <div className="min-h-screen bg-gray-900 flex items-center justify-center">
        <LoadingSpinner size="lg" />
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gray-900">
      {/* Header */}
      <div className="bg-gray-800 border-b border-gray-700">
        <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-6">
          <div className="flex items-center justify-between">
            <div>
              <h1 className="text-2xl font-bold text-gray-100">
                Panel de Súper Administrador
              </h1>
              <p className="mt-1 text-sm text-gray-400">
                Gestión de licencias y módulos del sistema
              </p>
            </div>
            <div className="flex items-center space-x-3">
              <span className="inline-flex items-center px-3 py-1 rounded-full text-xs font-medium bg-blue-900 text-blue-200">
                {activeCount} / {MODULES.length} módulos activos
              </span>
            </div>
          </div>
        </div>
      </div>

      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-8">
        {/* Messages */}
        <ErrorMessage error={error} className="mb-6" />
        
        {successMessage && (
          <div className="mb-6 bg-green-900/50 border border-green-700 text-green-200 px-4 py-3 rounded-lg">
            <p className="text-sm flex items-center">
              <svg className="w-5 h-5 mr-2" fill="currentColor" viewBox="0 0 20 20">
                <path fillRule="evenodd" d="M10 18a8 8 0 100-16 8 8 0 000 16zm3.707-9.293a1 1 0 00-1.414-1.414L9 10.586 7.707 9.293a1 1 0 00-1.414 1.414l2 2a1 1 0 001.414 0l4-4z" clipRule="evenodd" />
              </svg>
              {successMessage}
            </p>
          </div>
        )}

        {/* License Info Card */}
        {license && (
          <div className="bg-gray-800 rounded-xl border border-gray-700 p-6 mb-8">
            <div className="flex items-center justify-between">
              <div>
                <h2 className="text-lg font-semibold text-gray-100">
                  Licencia Empresarial
                </h2>
                <p className="text-sm text-gray-400 mt-1">
                  ID: {license.id.substring(0, 8)}...
                </p>
              </div>
              <div className="text-right">
                <p className="text-xs text-gray-500 uppercase tracking-wider">
                  Estado General
                </p>
                <p className="text-lg font-bold text-green-400 mt-1">
                  {activeCount === MODULES.length ? 'COMPLETA' : 'PERSONALIZADA'}
                </p>
              </div>
            </div>
          </div>
        )}

        {/* Module Toggles Grid */}
        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
          {MODULES.map((module) => {
            const isActive = license?.[module.key] ?? false;
            const isSaving = saving === module.key;

            return (
              <div
                key={module.key}
                className={`bg-gray-800 rounded-xl border transition-all duration-200 ${
                  isActive
                    ? 'border-blue-700 ring-1 ring-blue-700/50'
                    : 'border-gray-700'
                } ${isSaving ? 'opacity-70' : ''}`}
              >
                <div className="p-5">
                  <div className="flex items-center justify-between">
                    <div className="flex items-center space-x-3">
                      <span className="text-2xl">{module.icon}</span>
                      <div>
                        <h3 className="text-sm font-semibold text-gray-100">
                          {module.label}
                        </h3>
                        <p className="text-xs text-gray-400 mt-0.5">
                          {module.description}
                        </p>
                      </div>
                    </div>
                    <div className="flex items-center">
                      {isSaving ? (
                        <LoadingSpinner size="sm" />
                      ) : (
                        <button
                          type="button"
                          onClick={() => handleToggle(module.key)}
                          className={`relative inline-flex h-6 w-11 items-center rounded-full transition-colors duration-200 focus:outline-none focus:ring-2 focus:ring-blue-500 focus:ring-offset-2 focus:ring-offset-gray-800 ${
                            isActive ? 'bg-blue-600' : 'bg-gray-600'
                          }`}
                          aria-pressed={isActive}
                          aria-label={`${isActive ? 'Desactivar' : 'Activar'} ${module.label}`}
                        >
                          <span
                            className={`inline-block h-4 w-4 transform rounded-full bg-white transition-transform duration-200 ${
                              isActive ? 'translate-x-6' : 'translate-x-1'
                            }`}
                          />
                        </button>
                      )}
                    </div>
                  </div>

                  {/* Status indicators */}
                  <div className="mt-4 flex items-center space-x-4">
                    <span
                      className={`inline-flex items-center px-2 py-0.5 rounded text-xs font-medium ${
                        isActive
                          ? 'bg-green-900/50 text-green-300'
                          : 'bg-gray-700 text-gray-400'
                      }`}
                    >
                      <span
                        className={`w-1.5 h-1.5 rounded-full mr-1.5 ${
                          isActive ? 'bg-green-400' : 'bg-gray-500'
                        }`}
                      />
                      {isActive ? 'Activo' : 'Inactivo'}
                    </span>
                    {isActive && (
                      <span className="text-xs text-gray-500">
                        Consumiendo licencia
                      </span>
                    )}
                  </div>
                </div>
              </div>
            );
          })}
        </div>

        {/* Footer info */}
        <div className="mt-8 bg-gray-800/50 rounded-lg border border-gray-700/50 p-4">
          <p className="text-xs text-gray-500 text-center">
            Los cambios se aplican en tiempo real. Los módulos desactivados no estarán disponibles para ningún usuario, independientemente de su rol.
          </p>
        </div>
      </div>
    </div>
  );
};

export default SuperAdminPanel;
